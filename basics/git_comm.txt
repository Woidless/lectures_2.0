sudo apt install git - установка системы контроля версий

git --version - проверка версии

git config --global user.email "@.ru"
git config --global user.name "name"

git config --list - проверка настроек

ssh-keygen - создать ssh ключ

cat ~/.ssh/id/rsa.pub - получение ssh ключа

создаем репозитори на github

переходим в папку с проектом
git init - превращает текующие деректории в git репозиторий

git remote add название_связи(обычно - origin)ссылку_ на удаленный_репозиторий(SSH: git@github:)
git remote -v - просмотр списка удаленных репозиторий
git remote rm назв_связи (обычно - origin) - развязать связь
git status - проверка статуса файлов
git add название_файлов - отслеж. конкр. файлов
git add . - отслеживание всех файлов

git commit - созднание коммита (версии/слепка кода)
git commit -m 'Описание версии'

git push назв_связи(обычно origin) назв_ветки - отправка версии кода на удаленный репозиторий
Например: git push origin master

git commit -am ''

git clone ссылка_на-удаленный_репозиторий - склониров 

git diff  - просмотр внесенных изменений 

git branch - вывод списка веток в терминал
(звездочкой помечается активная ветка)

git branch название_ветки - создание ветки 

git checkout назв_ветки :
git switch назв_ветки - перход на указ ветку

git pull название_связи название_ветки - стягивание изменений с удаленного репозитория, с указанной ветки

git stash - спрятать локальные данные во временную область
git stash list - просмотр списка спрятанных данных
git stash apply - запись спряттанных данных/извлечение спрятанных данных
git stash drop - удаленные спрятанных данных

git log - просмотр коммитов/ версий
git log --oneline - компактное отображение коммитов

git remote -v
